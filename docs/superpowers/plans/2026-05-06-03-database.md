# Database & Migrations Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking. Each task follows TDD: write a failing test, run-fail, implement, run-pass, commit.

**Goal:** Stand up the persistent layer of СоциоПульс end-to-end: PgBouncer in transaction mode (Helm chart + ArgoCD app), `cmd/migrator` based on `golang-migrate`, the initial schema migration covering every business table from spec §6.3 with RLS-policies and the `tenancy_admin` BYPASSRLS role, a `pkg/postgres` pgx-pool wrapper that enforces `SET LOCAL app.tenant_id` per transaction, a `pkg/encryption` package implementing AES-256-GCM and HMAC-SHA256 phone hashing, an ArgoCD `PreSync` Job that runs migrations before every release, CI integration that runs migrations against an ephemeral `testcontainers-go` Postgres, and a smoke test that exercises down-migrations.

**Architecture:** `cmd/migrator` is a small CLI binary (`up`, `down`, `status`, `force`) wrapping `golang-migrate/v4` with the `pgx/v5` driver. Migrations live as numbered SQL pairs under `migrations/` (`000001_init.up.sql` + `000001_init.down.sql`). Application code never opens raw PostgreSQL connections — it goes through `pkg/postgres.Pool` which wraps `pgxpool.Pool` and `pkg/postgres.Tx` which always begins with `SET LOCAL app.tenant_id`. The `tenancy_admin` role bypasses RLS for service-owner level cross-tenant operations. PgBouncer is deployed as its own Helm chart (3 replicas, transaction mode, pool=25 per database, max_client=1000) with the upstream Yandex MPG cluster as the back end. ArgoCD reconciles a `Job` template with `argocd.argoproj.io/hook: PreSync` so migrations land before each `cmd/api` rollout. PII (phone numbers) is encrypted with AES-256-GCM (per-tenant DEK from KMS, cached in memory in subsequent plans) and indexed via a HMAC-SHA256 column with a per-tenant pepper.

**Tech Stack:** PostgreSQL 16 (Yandex Managed PG, provisioned in Plan 01), PgBouncer 1.22+, `github.com/jackc/pgx/v5` v5.6+, `github.com/jackc/pgx/v5/pgxpool`, `github.com/golang-migrate/migrate/v4` v4.17+, `github.com/testcontainers/testcontainers-go` v0.30+, `github.com/testcontainers/testcontainers-go/modules/postgres`, `github.com/stretchr/testify/require`. Build artefacts: `cmd/migrator` Docker image, PgBouncer Helm chart in `deployments/helm/pgbouncer/`, ArgoCD `Application` and `migrator` Job template in `deployments/argocd/applications/` and `deployments/helm/api/templates/`.

**Spec sections covered:** §6 (data model — every business table), §6.1 (RLS approach), §6.2 (PII encryption), §12.2 (RLS policy example), §12.3 (PgBouncer transaction mode), §12.4 (encryption at-rest), §16.5 (migrations workflow), §16.4 (CI/CD pipeline — `migration-dry-run`).

**Prerequisites:**
- Plan 00 completed (Go module `github.com/sociopulse/platform`, repo skeleton, Makefile, golangci-lint, GitHub Actions).
- Plan 01 completed (Yandex Managed PostgreSQL provisioned, `External Secrets Operator` deployed, `app-of-apps` ArgoCD root pointing at `deployments/argocd/applications/`).
- `kubectl` context set to the dev cluster (`yc managed-kubernetes cluster get-credentials sociopulse-dev`).
- Docker daemon available locally (testcontainers-go needs it for integration tests).

```go
import (

// repoRoot returns the absolute path of the repository root, computed from
// the test binary's working directory by walking up to find go.mod.
func repoRoot(t *testing.T) string {
	t.Helper()
	wd, err := os.Getwd()
	require.NoError(t, err)
	for dir := wd; dir != "/" && dir != "."; dir = filepath.Dir(dir) {
		if _, err := os.Stat(filepath.Join(dir, "go.mod")); err == nil {
			return dir
		}
	}
	t.Fatalf("could not locate go.mod from %s", wd)
	return ""
}

// startPostgres boots Postgres 16 in a container and returns its DSN.
```

```go
// Package main is the СоциоПульс database migration runner.
//
// Usage:
//
//	migrator up                          # apply all pending up migrations
//	migrator down                        # revert all migrations (dev/test)
//	migrator down --steps=1              # revert exactly one step
//	migrator status                      # print current version and dirty flag
//	migrator force <version>             # set version + clear dirty (recover)
//
// Environment:
//
//	DATABASE_URL       Postgres connection string (required)
//	MIGRATIONS_PATH    File URL pointing to the migrations directory
//	                   (default: file:///etc/sociopulse/migrations).
//
// Exit codes:
//
//	0  success
//	1  flag/usage error
//	2  migration error
//
// The binary is intentionally tiny: it does only what `golang-migrate`
// supports out of the box, plus structured logging suitable for k8s.
package main

import (
	"errors"
	"fmt"
	"io"
	"log/slog"
	"os"
	"strconv"

	"github.com/golang-migrate/migrate/v4"
	_ "github.com/golang-migrate/migrate/v4/database/pgx/v5"
	_ "github.com/golang-migrate/migrate/v4/source/file"
)

const defaultMigrationsPath = "file:///etc/sociopulse/migrations"

func main() {
	dsn := os.Getenv("DATABASE_URL")
	migPath := os.Getenv("MIGRATIONS_PATH")
	if migPath == "" {
		migPath = defaultMigrationsPath
	}
	if err := run(os.Args[1:], dsn, migPath, os.Stdout); err != nil {
		slog.Error("migrator failed", "err", err)
		var ue *usageError
		if errors.As(err, &ue) {
			fmt.Fprintln(os.Stderr, "usage: migrator <up|down|status|force> [args]")
			os.Exit(1)
		}
		os.Exit(2)
	}
}

type usageError struct{ msg string }

func (e *usageError) Error() string { return e.msg }

// run is the testable entrypoint. args is the slice without the program name.
func run(args []string, dsn, migrationsPath string, stdout io.Writer) error {
	if dsn == "" {
		return &usageError{msg: "DATABASE_URL is empty"}
	}
	if len(args) == 0 {
		return &usageError{msg: "no subcommand"}
	}

	m, err := migrate.New(migrationsPath, dsn)
	if err != nil {
		return fmt.Errorf("init migrate: %w", err)
	}
	defer func() {
		if srcErr, dbErr := m.Close(); srcErr != nil || dbErr != nil {
			slog.Warn("close migrate", "src_err", srcErr, "db_err", dbErr)
		}
	}()

	switch args[0] {
	case "up":
		if err := m.Up(); err != nil && !errors.Is(err, migrate.ErrNoChange) {
			return fmt.Errorf("up: %w", err)
		}
		return printVersion(m, stdout, "applied")

	case "down":
		steps := 0
		for _, a := range args[1:] {
			if v, ok := parseFlag(a, "--steps="); ok {
				n, err := strconv.Atoi(v)
				if err != nil || n <= 0 {
					return &usageError{msg: "invalid --steps value: " + v}
				}
				steps = n
			}
		}
		if steps > 0 {
			if err := m.Steps(-steps); err != nil && !errors.Is(err, migrate.ErrNoChange) {
				return fmt.Errorf("down %d steps: %w", steps, err)
			}
		} else {
			if err := m.Down(); err != nil && !errors.Is(err, migrate.ErrNoChange) {
				return fmt.Errorf("down all: %w", err)
			}
		}
		return printVersion(m, stdout, "reverted")

	case "status":
		return printVersion(m, stdout, "current")

	case "force":
		if len(args) < 2 {
			return &usageError{msg: "force requires <version>"}
		}
		v, err := strconv.Atoi(args[1])
		if err != nil {
			return &usageError{msg: "force: version must be an integer"}
		}
		if err := m.Force(v); err != nil {
			return fmt.Errorf("force %d: %w", v, err)
		}
		return printVersion(m, stdout, "forced")

	default:
		return &usageError{msg: "unknown subcommand: " + args[0]}
	}
}

func printVersion(m *migrate.Migrate, w io.Writer, prefix string) error {
	v, dirty, err := m.Version()
	if errors.Is(err, migrate.ErrNilVersion) {
		fmt.Fprintf(w, "%s: version=none dirty=false\n", prefix)
		return nil
	}
	if err != nil {
		return fmt.Errorf("version: %w", err)
	}
	fmt.Fprintf(w, "%s: version=%d dirty=%t\n", prefix, v, dirty)
	return nil
}

func parseFlag(arg, prefix string) (string, bool) {
	if len(arg) < len(prefix) || arg[:len(prefix)] != prefix {
		return "", false
	}
	return arg[len(prefix):], true
}
```

```sql
      'operator_sessions','operator_state_log',
      'audit_log','reports_jobs'

    execute format('alter table %I enable row level security', t);
    execute format('alter table %I force row level security', t);
  end loop;
end$$;
```

## Task 4: `pkg/postgres` — pgxpool wrapper with `SET LOCAL app.tenant_id`

**Goal:** Build the application-side connection layer. `pkg/postgres.Pool` wraps `pgxpool.Pool` with a constructor that applies sane defaults (max conns, health checks). Most importantly, it provides `WithTenant(ctx, tenantID, fn)` which:
1. Begins a transaction.
2. Executes `SET LOCAL app.tenant_id = '<uuid>'`.
3. Calls the user-supplied function with a `Tx` value.
4. Commits or rolls back.

This guarantees that no application code can issue a query without first setting the tenant — RLS enforces isolation if it ever happens, but the wrapper makes the right path the easy path. We also expose `BypassRLS(ctx, fn)` for the `tenancy` module: it does `SET LOCAL ROLE tenancy_admin` instead.

```go
package postgres

import (
	"context"
	"errors"
	"fmt"
	"time"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5"
	"github.com/jackc/pgx/v5/pgconn"
	"github.com/jackc/pgx/v5/pgxpool"
)

// Config configures the Postgres connection pool. Most callers use
// Plan-02-defined `config.Postgres` and copy fields over.
type Config struct {
	// DSN is the libpq-style connection string, typically pointing at
	// PgBouncer ("postgres://app:pwd@pgbouncer:6432/sociopulse?sslmode=verify-full").
	DSN string

	// MaxConns is the upper bound on connections in the pool. Should be
	// chosen with PgBouncer's default_pool_size in mind: cluster-wide,
	// total-Go-pool-conns ≤ pgbouncer-default-pool-size × replicas.
	MaxConns int32

	// MinConns is the warm pool size. 1 is fine for dev; production usually
	// matches PgBouncer's min_pool_size.
	MinConns int32

	// ConnectTimeout caps individual connection-open attempts.
	ConnectTimeout time.Duration

	// HealthCheckPeriod is how often pgxpool pings idle conns. 0 → pgx default.
	HealthCheckPeriod time.Duration
}

// Pool is the application-facing pool. It does not expose pgxpool.Pool
// directly so we control how transactions begin.
type Pool struct {
	p *pgxpool.Pool
}

// Open initialises the pool. It applies SETUP statements once per
// new connection via AfterConnect (currently a no-op; placeholders remain
// for future timezone / search_path tweaks).
func Open(ctx context.Context, c Config) (*Pool, error) {
	if c.DSN == "" {
		return nil, errors.New("postgres: empty DSN")
	}
	cfg, err := pgxpool.ParseConfig(c.DSN)
	if err != nil {
		return nil, fmt.Errorf("postgres: parse DSN: %w", err)
	}
	if c.MaxConns > 0 {
		cfg.MaxConns = c.MaxConns
	}
	if c.MinConns >= 0 {
		cfg.MinConns = c.MinConns
	}
	if c.ConnectTimeout > 0 {
		cfg.ConnConfig.ConnectTimeout = c.ConnectTimeout
	}
	if c.HealthCheckPeriod > 0 {
		cfg.HealthCheckPeriod = c.HealthCheckPeriod
	}
	cfg.AfterConnect = func(ctx context.Context, conn *pgx.Conn) error {
		// No-op: leave hook in place for future startup queries
		// (timezone, search_path, plan-cache size...).
		return nil
	}
	pool, err := pgxpool.NewWithConfig(ctx, cfg)
	if err != nil {
		return nil, fmt.Errorf("postgres: connect: %w", err)
	}
	return &Pool{p: pool}, nil
}

// Close drains and closes the pool. It is idempotent.
func (p *Pool) Close() {
	if p == nil || p.p == nil {
		return
	}
	p.p.Close()
}

// Ping verifies the pool can talk to Postgres. Used for /readyz handlers.
func (p *Pool) Ping(ctx context.Context) error {
	return p.p.Ping(ctx)
}

// WithTenant runs fn inside a transaction with `SET LOCAL app.tenant_id`.
// On nil error from fn, the tx commits; on any error, it rolls back.
//
// The tenant id is rendered safely via PostgreSQL's `set_config` builtin
// rather than string concatenation, because `SET LOCAL` does not accept
// parameterised values.
func (p *Pool) WithTenant(ctx context.Context, tenantID uuid.UUID, fn func(Tx) error) error {
	if tenantID == uuid.Nil {
		return errors.New("postgres: WithTenant requires a non-zero tenant id")
	}
	return p.transact(ctx, func(tx pgx.Tx) error {
		if _, err := tx.Exec(ctx,
			"select set_config('app.tenant_id', $1::text, true)", tenantID.String()); err != nil {
			return fmt.Errorf("postgres: set tenant: %w", err)
		}
		return fn(Tx{tx: tx})
	})
}

// BypassRLS runs fn inside a transaction whose role is `tenancy_admin`,
// which has BYPASSRLS. Reserved for the tenancy module.
func (p *Pool) BypassRLS(ctx context.Context, fn func(Tx) error) error {
	return p.transact(ctx, func(tx pgx.Tx) error {
		if _, err := tx.Exec(ctx, "set local role tenancy_admin"); err != nil {
			return fmt.Errorf("postgres: set role: %w", err)
		}
		return fn(Tx{tx: tx})
	})
}

// RawExec is an unscoped Exec for migrations and for testing only. Application
// code MUST go through WithTenant or BypassRLS.
func (p *Pool) RawExec(ctx context.Context, sql string, args ...any) (pgconn.CommandTag, error) {
	return p.p.Exec(ctx, sql, args...)
}

// RawQueryRow is similarly only for boot-time and tests.
func (p *Pool) RawQueryRow(ctx context.Context, sql string, args ...any) pgx.Row {
	return p.p.QueryRow(ctx, sql, args...)
}

func (p *Pool) transact(ctx context.Context, fn func(pgx.Tx) error) error {
	tx, err := p.p.BeginTx(ctx, pgx.TxOptions{})
	if err != nil {
		return fmt.Errorf("postgres: begin: %w", err)
	}
	defer func() {
		// Rollback is a no-op if Commit succeeded.
		_ = tx.Rollback(context.Background())
	}()

	if err := fn(tx); err != nil {
		return err
	}
	if err := tx.Commit(ctx); err != nil {
		return fmt.Errorf("postgres: commit: %w", err)
	}
	return nil
}
```

```go
package postgres

import (
	"context"

	"github.com/jackc/pgx/v5"
	"github.com/jackc/pgx/v5/pgconn"
)

// Tx is the transaction handle exposed to store-layer code. It wraps pgx.Tx
// without exposing it directly, so callers can't accidentally Commit/Rollback
// (the wrapper owns lifecycle).
type Tx struct {
	tx pgx.Tx
}

// Exec executes a SQL statement that returns no rows.
func (t Tx) Exec(ctx context.Context, sql string, args ...any) (pgconn.CommandTag, error) {
	return t.tx.Exec(ctx, sql, args...)
}

// Query runs a multi-row query. Callers MUST close the returned Rows.
func (t Tx) Query(ctx context.Context, sql string, args ...any) (pgx.Rows, error) {
	return t.tx.Query(ctx, sql, args...)
}

// QueryRow runs a single-row query.
func (t Tx) QueryRow(ctx context.Context, sql string, args ...any) pgx.Row {
	return t.tx.QueryRow(ctx, sql, args...)
}

// CopyFrom delegates to pgx.Tx.CopyFrom for bulk inserts (used by import flows).
func (t Tx) CopyFrom(ctx context.Context, table pgx.Identifier, columns []string, rows pgx.CopyFromSource) (int64, error) {
	return t.tx.CopyFrom(ctx, table, columns, rows)
}

// Conn returns the underlying connection — needed by some pgx helpers
// (e.g. listening for notifications). Use sparingly.
func (t Tx) Conn() *pgx.Conn {
	return t.tx.Conn()
}
```

```go
package encryption

import (
	"crypto/aes"
	"crypto/cipher"
	"crypto/rand"
	"errors"
	"fmt"
	"io"
)

// ErrAuth is returned by Decrypt when the GCM authentication check fails:
// either the ciphertext was tampered with, or the additionalData doesn't
// match what was supplied during Encrypt.
var ErrAuth = errors.New("encryption: authentication failed")

// ErrKeySize is returned by Encrypt/Decrypt when the key is not 32 bytes
// (we lock to AES-256 only — the spec is unambiguous on this).
var ErrKeySize = errors.New("encryption: key must be 32 bytes (AES-256)")

const (
	keyLen   = 32
	nonceLen = 12
)

// Encrypt performs AES-256-GCM and returns nonce || ciphertext || tag.
//
// additionalData is bound by GCM but not stored in the output. Common
// choices: tenant id, table name, column name — anything that the
// caller can reproduce at Decrypt time and that uniquely identifies the
// row. Pass nil for "no AAD".
func Encrypt(key, plaintext, additionalData []byte) ([]byte, error) {
	if len(key) != keyLen {
		return nil, fmt.Errorf("%w: got %d", ErrKeySize, len(key))
	}
	block, err := aes.NewCipher(key)
	if err != nil {
		return nil, fmt.Errorf("encryption: new cipher: %w", err)
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return nil, fmt.Errorf("encryption: new gcm: %w", err)
	}
	nonce := make([]byte, nonceLen)
	if _, err := io.ReadFull(rand.Reader, nonce); err != nil {
		return nil, fmt.Errorf("encryption: read nonce: %w", err)
	}
	// Seal appends to nonce so the nonce ends up as the prefix.
	out := gcm.Seal(nonce, nonce, plaintext, additionalData)
	return out, nil
}

// Decrypt reverses Encrypt. additionalData must match exactly what was
// passed to Encrypt; mismatch yields ErrAuth (wrapped through GCM's auth
// check).
func Decrypt(key, ciphertext, additionalData []byte) ([]byte, error) {
	if len(key) != keyLen {
		return nil, fmt.Errorf("%w: got %d", ErrKeySize, len(key))
	}
	if len(ciphertext) < nonceLen+16 { // nonce + at least one tag block
		return nil, errors.New("encryption: ciphertext too short")
	}
	block, err := aes.NewCipher(key)
	if err != nil {
		return nil, fmt.Errorf("encryption: new cipher: %w", err)
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return nil, fmt.Errorf("encryption: new gcm: %w", err)
	}
	nonce, body := ciphertext[:nonceLen], ciphertext[nonceLen:]
	plaintext, err := gcm.Open(nil, nonce, body, additionalData)
	if err != nil {
		return nil, fmt.Errorf("%w: %v", ErrAuth, err)
	}
	return plaintext, nil
}
```

```go
package encryption

import (
	"crypto/hmac"
	"crypto/sha256"
	"strings"
)

// PhoneHasher computes deterministic HMAC-SHA256 hashes of phone numbers
// using a per-tenant pepper. It is safe for concurrent use; the pepper
// is captured by reference and never mutated.
type PhoneHasher struct {
	pepper []byte
}

// NewPhoneHasher constructs a PhoneHasher. The pepper MUST be at least
// 32 bytes (the spec specifies "random 32 bytes per-tenant"). Shorter
// peppers panic — there's no recoverable scenario where a too-short
// pepper is intentional.
func NewPhoneHasher(pepper []byte) *PhoneHasher {
	if len(pepper) < 32 {
		panic("encryption: phone hash pepper must be ≥ 32 bytes")
	}
	cp := make([]byte, len(pepper))
	copy(cp, pepper)
	return &PhoneHasher{pepper: cp}
}

// Hash returns the 32-byte HMAC-SHA256 of the normalised phone number.
func (h *PhoneHasher) Hash(phone string) []byte {
	mac := hmac.New(sha256.New, h.pepper)
	mac.Write([]byte(NormalizePhone(phone)))
	return mac.Sum(nil)
}

// NormalizePhone canonicalises Russian phone numbers to E.164 form
// "+7XXXXXXXXXX". It strips spaces, dashes, parentheses, leading "8" or
// "7" → "+7", and validates the length softly. Inputs that don't match
// any Russian-pattern fall through unchanged so the caller can detect
// downstream (we don't fail-noisy here because the hasher is the wrong
// place to enforce input validation — that's the import-pipeline job).
func NormalizePhone(phone string) string {
	digits := strings.Builder{}
	digits.Grow(len(phone))
	for _, r := range phone {
		if r >= '0' && r <= '9' {
			digits.WriteRune(r)
		}
	}
	d := digits.String()
	switch {
	case len(d) == 11 && d[0] == '8':
		return "+7" + d[1:]
	case len(d) == 11 && d[0] == '7':
		return "+7" + d[1:]
	case len(d) == 10:
		return "+7" + d
	default:
		return phone
	}
}
```

```go
package outbox

import (
    "context"
    "encoding/json"
    "fmt"

    "github.com/jackc/pgx/v5"
    "github.com/jackc/pgx/v5/pgxpool"
)

// PostgresWriter inserts events into the event_outbox table.
//
// If a pgx.Tx is present in ctx (via WithTx), the insert joins the caller's
// transaction — this is the canonical way to use the outbox: write business
// state and outbox row atomically.
type PostgresWriter struct {
    pool *pgxpool.Pool
}

func NewPostgresWriter(pool *pgxpool.Pool) *PostgresWriter {
    return &PostgresWriter{pool: pool}
}

type txKey struct{}

// WithTx returns a context that pins outbox writes to the given transaction.
func WithTx(ctx context.Context, tx pgx.Tx) context.Context {
    return context.WithValue(ctx, txKey{}, tx)
}

func (w *PostgresWriter) Append(ctx context.Context, e Event) error {
    payload, err := json.Marshal(e.Payload)
    if err != nil {
        return fmt.Errorf("marshal payload: %w", err)
    }
    if e.Subject == "" {
        return fmt.Errorf("outbox: empty subject")
    }

    const q = `INSERT INTO event_outbox(tenant_id, aggregate_id, subject, payload)
               VALUES ($1, $2, $3, $4)`

    if tx, ok := ctx.Value(txKey{}).(pgx.Tx); ok {
        _, err = tx.Exec(ctx, q, e.TenantID, e.AggregateID, e.Subject, payload)
    } else {
        _, err = w.pool.Exec(ctx, q, e.TenantID, e.AggregateID, e.Subject, payload)
    }
    if err != nil {
        return fmt.Errorf("insert outbox row: %w", err)
    }
    return nil
}
```

```go
package outbox

import (
    "context"
    "encoding/json"
    "fmt"
    "time"

    "github.com/jackc/pgx/v5"
    "github.com/jackc/pgx/v5/pgxpool"
    "go.uber.org/zap"
)

// Publisher is the destination interface — typically a NATS-backed publisher
// from internal/realtime, but kept abstract so unit tests can use a fake.
type Publisher interface {
    Publish(ctx context.Context, subject string, payload []byte) error
}

type RelayConfig struct {
    BatchSize int           // max rows per drain (default 100)
    Tick      time.Duration // poll interval (default 1s)
    MaxRetry  int           // max attempts before parking the row (default 10)
    Logger    *zap.Logger
}

func (c *RelayConfig) defaults() {
    if c.BatchSize == 0 {
        c.BatchSize = 100
    }
    if c.Tick == 0 {
        c.Tick = time.Second
    }
    if c.MaxRetry == 0 {
        c.MaxRetry = 10
    }
    if c.Logger == nil {
        c.Logger = zap.NewNop()
    }
}

type Relay struct {
    pool *pgxpool.Pool
    pub  Publisher
    cfg  RelayConfig
}

func NewRelay(pool *pgxpool.Pool, pub Publisher, cfg RelayConfig) *Relay {
    cfg.defaults()
    return &Relay{pool: pool, pub: pub, cfg: cfg}
}

// Run loops until ctx is cancelled. Errors are logged; the loop never returns
// while ctx is alive — outbox-relay is a critical infrastructure goroutine.
func (r *Relay) Run(ctx context.Context) {
    t := time.NewTicker(r.cfg.Tick)
    defer t.Stop()
    for {
        select {
        case <-ctx.Done():
            return
        case <-t.C:
            if err := r.drainOnce(ctx); err != nil {
                r.cfg.Logger.Error("outbox drain failed", zap.Error(err))
            }
        }
    }
}

func (r *Relay) drainOnce(ctx context.Context) error {
    rows, err := r.pool.Query(ctx, `
        SELECT id, subject, payload, attempts
        FROM event_outbox
        WHERE published_at IS NULL AND attempts < $2
        ORDER BY created_at
        LIMIT $1
        FOR UPDATE SKIP LOCKED`,
        r.cfg.BatchSize, r.cfg.MaxRetry)
    if err != nil {
        return fmt.Errorf("select pending: %w", err)
    }
    defer rows.Close()

    type pending struct {
        id      int64
        subject string
        payload []byte
    }
    var batch []pending
    for rows.Next() {
        var p pending
        if err := rows.Scan(&p.id, &p.subject, &p.payload, new(int)); err != nil {
            return fmt.Errorf("scan row: %w", err)
        }
        batch = append(batch, p)
    }
    if err := rows.Err(); err != nil {
        return err
    }
    rows.Close()

    for _, p := range batch {
        if err := r.publishAndMark(ctx, p.id, p.subject, p.payload); err != nil {
            r.cfg.Logger.Warn("outbox publish failed; will retry",
                zap.Int64("id", p.id), zap.String("subject", p.subject), zap.Error(err))
        }
    }
    return nil
}

func (r *Relay) publishAndMark(ctx context.Context, id int64, subject string, payload []byte) error {
    // Validate JSON payload by round-tripping (cheap defence).
    var probe map[string]any
    if err := json.Unmarshal(payload, &probe); err != nil {
        // Park the row by maxing attempts so future drains skip it; alert via metric.
        _, _ = r.pool.Exec(ctx,
            `UPDATE event_outbox SET attempts = $2, last_error = $3 WHERE id = $1`,
            id, r.cfg.MaxRetry, "invalid payload json")
        return fmt.Errorf("invalid payload for id=%d: %w", id, err)
    }

    pubCtx, cancel := context.WithTimeout(ctx, 5*time.Second)
    defer cancel()
    if err := r.pub.Publish(pubCtx, subject, payload); err != nil {
        _, _ = r.pool.Exec(ctx,
            `UPDATE event_outbox SET attempts = attempts + 1, last_error = $2 WHERE id = $1`,
            id, err.Error())
        return err
    }

    _, err := r.pool.Exec(ctx,
        `UPDATE event_outbox SET published_at = NOW(), last_error = NULL WHERE id = $1`, id)
    return err
}
```

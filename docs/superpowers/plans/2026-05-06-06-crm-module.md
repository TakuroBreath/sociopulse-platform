# CRM Module Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. REQUIRED METHOD: superpowers:test-driven-development on every step that touches code. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `crm` domain module: project lifecycle (create, list, update, pause/resume/archive, progress aggregation), respondent CRUD with russian-format phone validation/normalization, async CSV/XLSX import with deduplication and progress events, real-time quota tracker with Redis cache and a Postgres reconciliation worker, per-tenant + per-project DNC with import, the deletion right (FR-§13.3) implemented as 30-day soft delete + `worker.respondents.purge`, and HTTP endpoints exposed under `/api/projects/*` and `/api/respondents/*`. All mutations append to `audit_log`. Service-layer behavior is fully covered by unit tests with `gomock`, integration tests use `testcontainers-go` for real Postgres + Redis, and a load test imports 100 000 respondents end-to-end in under 60 seconds.

**Architecture:** The module follows the standard layout established in Plan 00:

```
internal/crm/
├── api/         # public Go interfaces + DTOs (importable cross-module)
├── service/     # business logic (private)
├── store/       # pgx-based Postgres access (private)
├── events/      # NATS publishers + asynq task definitions (private)
└── http/        # gin handlers, route binding (private)
```

`api` exposes `ProjectService`, `RespondentService`, `QuotaTracker`, `DNCManager` interfaces. `service` provides concrete implementations wired via `New*Service(deps)` constructors. `store` holds raw SQL — no ORM. `events` defines the asynq task types `crm:respondent.import`, `crm:respondents.purge`, `crm:quotas.recompute`, plus NATS subjects `crm.project.<event>` and `crm.respondents.import.<event>`. `http` mounts handlers onto a `gin.RouterGroup` provided by `gateway`.

The dialer (Plan 09) calls `QuotaTracker.IsFull` on the hot path before each ZPOPMIN — the cache makes this an in-process lookup most of the time. Increment is transactional and runs inside the same Postgres transaction that finalises a successful call.

**Tech Stack:** Go 1.26+, `github.com/jackc/pgx/v5`, `github.com/redis/go-redis/v9`, `github.com/hibiken/asynq` v0.24+, `github.com/xuri/excelize/v2` v2.8+, `github.com/google/uuid`, `github.com/stretchr/testify`, `go.uber.org/mock` (gomock), `github.com/testcontainers/testcontainers-go` v0.31+, `github.com/gin-gonic/gin`, `github.com/nats-io/nats.go`.

```go
package api

import (

// ProjectStatus is the lifecycle state of a project.
type ProjectStatus string

const (
	StatusActive   ProjectStatus = "active"
	StatusPaused   ProjectStatus = "paused"
	StatusArchived ProjectStatus = "archived"
)

// Project is the read-side DTO of a project row.
type Project struct {
	ID                       uuid.UUID
	TenantID                 uuid.UUID
	Code                     string
	Name                     string
	Customer                 string
	Status                   ProjectStatus
	TargetCount              int
	PeriodFrom               *time.Time
	PeriodTo                 *time.Time
	SurveyID                 *uuid.UUID
	DefaultSurveyVersionID   *uuid.UUID
	IsAdvertising            bool
	CreatedAt                time.Time
	Quotas                   []Quota          // optional, filled by Get
	Assignments              []ProjectMember  // optional, filled by Get
}

// CreateProjectInput is the payload for ProjectService.Create.
type CreateProjectInput struct {
	Code           string
	Name           string
	Customer       string
	TargetCount    int
	PeriodFrom     *time.Time
	PeriodTo       *time.Time
	SurveyID       *uuid.UUID
	IsAdvertising  bool
	InitialQuotas  []Quota          // optional batch of project_quotas rows
	InitialMembers []uuid.UUID      // optional operator IDs to assign
}

// UpdateProjectInput is the payload for ProjectService.Update — all fields optional.
type UpdateProjectInput struct {
	Name        *string
	Customer    *string
	TargetCount *int
	PeriodFrom  *time.Time
	PeriodTo    *time.Time
	SurveyID    *uuid.UUID
}

// ProjectMember is an operator assignment row.
type ProjectMember struct {
	OperatorID uuid.UUID
	AssignedAt time.Time
}

// ListProjectsFilter narrows a List() call.
type ListProjectsFilter struct {
	Status   *ProjectStatus
	Search   string // substring match against name/code/customer
	Page     int    // 1-based
	PageSize int    // capped at 200
}

// ListProjectsResult is the page returned by List.
type ListProjectsResult struct {
	Items      []Project
	TotalCount int
}

// ProjectProgress is the aggregated progress snapshot.
type ProjectProgress struct {
	ProjectID         uuid.UUID
	TargetCount       int
	CompletedCount    int
	InProgressCount   int
	PendingCount      int
	DNCCount          int
	ExhaustedCount    int
	WrongCount        int
	PercentDone       float64        // 0..100
	PaceLast24h       int            // completed in the last 24h
	ETACompletion     *time.Time     // nil if pace==0
	QuotaProgress     []QuotaSnapshot
}

// QuotaSnapshot is one (kind, value) line from project_quotas.

	DimensionKind  string
	DimensionValue string
	Target         int
	Done           int
	PercentDone    float64
	IsFull         bool
}

// ProjectService is the public contract for the project lifecycle.

	Create(ctx context.Context, in CreateProjectInput) (*Project, error)
	Get(ctx context.Context, id uuid.UUID) (*Project, error)

	Resume(ctx context.Context, id uuid.UUID) error
	Archive(ctx context.Context, id uuid.UUID) error
	GetProgress(ctx context.Context, id uuid.UUID) (*ProjectProgress, error)
	Assign(ctx context.Context, id uuid.UUID, operatorIDs []uuid.UUID) error
	Unassign(ctx context.Context, id uuid.UUID, operatorID uuid.UUID) error
	ListMembers(ctx context.Context, id uuid.UUID) ([]ProjectMember, error)
}
```

```go
package store

import (
	"context"

	crmapi "github.com/sociopulse/platform/internal/crm/api"

// DBTX is the minimal pgx contract — pool or tx.
type DBTX interface {
	Exec(ctx context.Context, sql string, args ...any) (pgconn.CommandTag, error)
	Query(ctx context.Context, sql string, args ...any) (pgx.Rows, error)
	QueryRow(ctx context.Context, sql string, args ...any) pgx.Row
}

// ProjectRow mirrors the projects table.
type ProjectRow struct {
	ID                     uuid.UUID
	TenantID               uuid.UUID
	Code                   string
	Name                   string
	Customer               string
	Status                 string
	TargetCount            int
	PeriodFrom             *time.Time
	PeriodTo               *time.Time
	SurveyID               *uuid.UUID
	DefaultSurveyVersionID *uuid.UUID
	IsAdvertising          bool
	CreatedAt              time.Time
}

// ProjectStore is the persistence contract used by service.ProjectService.
//
//go:generate mockgen -source=project.go -destination=../service/mocks/project_store_mock.go -package=mocks ProjectStore
type ProjectStore interface {
	WithTx(ctx context.Context, fn func(tx DBTX) error) error
	Insert(ctx context.Context, row *ProjectRow) error
	Get(ctx context.Context, tenantID, id uuid.UUID) (*ProjectRow, error)
	GetByCode(ctx context.Context, tenantID uuid.UUID, code string) (*ProjectRow, error)
	List(ctx context.Context, tenantID uuid.UUID, f crmapi.ListProjectsFilter) ([]ProjectRow, int, error)
	Update(ctx context.Context, tenantID uuid.UUID, id uuid.UUID, patch ProjectPatch) error
	UpdateStatus(ctx context.Context, tenantID, id uuid.UUID, from, to crmapi.ProjectStatus) error
	InsertQuotas(ctx context.Context, tenantID, projectID uuid.UUID, quotas []crmapi.Quota) error
	ListQuotas(ctx context.Context, tenantID, projectID uuid.UUID) ([]crmapi.QuotaSnapshot, error)
	InsertAssignments(ctx context.Context, tenantID, projectID uuid.UUID, ops []uuid.UUID) error
	DeleteAssignment(ctx context.Context, tenantID, projectID, opID uuid.UUID) error
	ListAssignments(ctx context.Context, tenantID, projectID uuid.UUID) ([]crmapi.ProjectMember, error)
	AggregateRespondentStatuses(ctx context.Context, tenantID, projectID uuid.UUID) (map[string]int, error)
	CompletedSince(ctx context.Context, tenantID, projectID uuid.UUID, since time.Time) (int, error)
}

// ProjectPatch is the diff for Update.
type ProjectPatch struct {
	Name        *string
	Customer    *string
	TargetCount *int
	PeriodFrom  *time.Time
	PeriodTo    *time.Time
	SurveyID    *uuid.UUID
}

// New creates a pgx-backed store.
func NewProjectStore(pool *pgxpool.Pool) ProjectStore {
	return &projectStore{pool: pool}
}

type projectStore struct {
	pool *pgxpool.Pool
}

func (s *projectStore) WithTx(ctx context.Context, fn func(tx DBTX) error) error {
	tx, err := s.pool.BeginTx(ctx, pgx.TxOptions{})
	if err != nil {
		return err
	}
	defer tx.Rollback(ctx)
	if err := fn(tx); err != nil {
		return err
	}
	return tx.Commit(ctx)
}
```

```go
func auditPayload(actor uuid.UUID, fields map[string]any) map[string]any {
	out := map[string]any{"actor_id": actor}
	for k, v := range fields {
		out[k] = v
```

```go
func NewProjectService(d ProjectDeps) *ProjectService {

// Create persists a project and emits the corresponding audit/event.
func (s *ProjectService) Create(ctx context.Context, in crmapi.CreateProjectInput) (*crmapi.Project, error) {
	tenantID, actorID, err := tenantAndActor(ctx)
	if err != nil {
		return nil, err

		CreatedAt:     s.deps.Clock().UTC(),

			return err

			}

				return err
			}
		}
		return nil
	})

		return nil, err
	}

	_ = s.deps.Audit.Write(ctx, "project.created", auditPayload(actorID, map[string]any{
		"project_id": row.ID, "code": row.Code, "name": row.Name,
	}))
```

```go
package service

import (
	"errors"
	"strings"
	"unicode"

	crmapi "github.com/sociopulse/platform/internal/crm/api"
)

// NormalizePhone converts any acceptable Russian phone form into E.164 +7XXXXXXXXXX
// and returns the 3-digit region/operator code. Invalid input returns ErrInvalidPhone.
//
// Accepted inputs:
//   - +7XXXXXXXXXX           (already canonical)
//   - 7XXXXXXXXXX            (no plus)
//   - 8XXXXXXXXXX            (Russian internal long-distance prefix)
//   - any of the above with spaces, dashes, parentheses, dots
func NormalizePhone(in string) (canonical, region string, err error) {
	digits := stripNonDigits(in)
	if digits == "" {
		return "", "", crmapi.ErrInvalidPhone
	}
	switch {
	case strings.HasPrefix(digits, "8") && len(digits) == 11:
		digits = "7" + digits[1:]
	case strings.HasPrefix(digits, "+7"):
		digits = digits[1:] // drop '+'
	}
	// At this point digits should be 11 chars starting with 7.
	if len(digits) != 11 || digits[0] != '7' {
		return "", "", crmapi.ErrInvalidPhone
	}
	canonical = "+" + digits
	region = digits[1:4] // 3 digits after country code
	return canonical, region, nil
}

// MaskPhone returns a UI-safe mask hiding 7 of the 10 subscriber digits.
func MaskPhone(p string) string {
	if len(p) < 4 {
		return "***"
	}
	// Expect "+7XXXXXXXXXX" — 12 chars.
	if !strings.HasPrefix(p, "+7") || len(p) != 12 {
		return "***"
	}
	digit3 := p[2:3] // first digit of region (kept)
	last2 := p[10:12]
	return "+7-" + digit3 + "**-***-**-" + last2
}

func stripNonDigits(s string) string {
	var b strings.Builder
	for _, r := range s {
		if unicode.IsDigit(r) || r == '+' {
			b.WriteRune(r)
		}
	}
	return b.String()
}

var _ = errors.New // ensure import retained when refactored
```

```go
func (s *respStore) Search(ctx context.Context, tenantID uuid.UUID, f crmapi.SearchRespondentsFilter, hash []byte) ([]RespondentRow, int, error) {
	args := []any{tenantID, f.ProjectID}
	where := []string{"tenant_id = $1", "project_id = $2"}
	if f.Status != nil {
		args = append(args, string(*f.Status))
		where = append(where, fmtIdx("status = $%d", len(args)))
	}
	if hash != nil {
		args = append(args, hash)
		where = append(where, fmtIdx("phone_hash = $%d", len(args)))
	}
	if f.Region != "" {
		args = append(args, f.Region)
		where = append(where, fmtIdx("region_code = $%d", len(args)))
	}
	clause := joinWhere(where)
	var total int
	if err := s.pool.QueryRow(ctx, "select count(*) from respondents "+clause, args...).Scan(&total); err != nil {
		return nil, 0, err
	}
	args = append(args, f.PageSize, (f.Page-1)*f.PageSize)
	rows, err := s.pool.Query(ctx,
		"select id, tenant_id, project_id, region_code, attributes, status, attempts, last_attempt_at, next_attempt_at, source, created_at "+
			"from respondents "+clause+" order by created_at desc limit $"+itoa(len(args)-1)+" offset $"+itoa(len(args)),
		args...)
	if err != nil {
		return nil, 0, err
	}
	defer rows.Close()
	out := []RespondentRow{}
	for rows.Next() {
		var r RespondentRow
		if err := rows.Scan(&r.ID, &r.TenantID, &r.ProjectID, &r.RegionCode, &r.Attributes, &r.Status,
			&r.Attempts, &r.LastAttemptAt, &r.NextAttemptAt, &r.Source, &r.CreatedAt); err != nil {
			return nil, 0, err
		}
		out = append(out, r)
	}
	return out, total, rows.Err()
}

func (s *respStore) MarkDeletionRequested(ctx context.Context, tenantID, id uuid.UUID, deleteAt time.Time) error {
	tag, err := s.pool.Exec(ctx,
		`update respondents set status = 'deletion-requested', delete_at = $3 where tenant_id = $1 and id = $2`,
		tenantID, id, deleteAt)
	if err != nil {
		return err
	}
	if tag.RowsAffected() == 0 {
		return ErrNoRows
	}
	return nil
}

func (s *respStore) PurgeDue(ctx context.Context, now time.Time, batch int) (int, error) {
	tag, err := s.pool.Exec(ctx, `
		update respondents
		set phone_encrypted = '\x00'::bytea, phone_hash = '\x00'::bytea,
		    attributes = '{}'::jsonb, region_code = '', status = 'completed', delete_at = null
		where status = 'deletion-requested' and delete_at <= $1
		and id in (select id from respondents
		           where status = 'deletion-requested' and delete_at <= $1
		           order by delete_at limit $2)`,
		now, batch)
	if err != nil {
		return 0, err
	}
	return int(tag.RowsAffected()), nil
}

func fmtIdx(format string, idx int) string {
	return fmt.Sprintf(format, idx)
}
func joinWhere(w []string) string {
	if len(w) == 0 {
		return ""
	}
	return "where " + strings.Join(w, " and ")
}
```

```go
// Create persists a single respondent (used by UI single-add).
func (s *RespondentService) Create(ctx context.Context, in crmapi.CreateRespondentInput) (*crmapi.Respondent, error) {
	tenantID, actor, err := tenantAndActor(ctx)
	if err != nil {
		return nil, err
	}
	canonical, region, err := NormalizePhone(in.Phone)
	if err != nil {
		return nil, err
	}
	if in.RegionCode != "" {
		region = in.RegionCode
	}
	hash, err := s.deps.Crypto.HashPhone(ctx, canonical)
	if err != nil {
		return nil, fmt.Errorf("hash phone: %w", err)
	}
	blocked, err := s.deps.DNC.IsBlocked(ctx, tenantID, actor, &in.ProjectID, hash)
	if err != nil {
		return nil, err
	}
	if blocked {
		return nil, crmapi.ErrPhoneInDNC
	}
	if existing, err := s.deps.Store.LookupByHash(ctx, tenantID, in.ProjectID, hash); err != nil && !errors.Is(err, store.ErrNoRows) {
		return nil, err
	} else if existing != nil {
		return nil, crmapi.ErrDuplicateRespondent
	}
	cipher, err := s.deps.Crypto.EncryptPhone(ctx, canonical)
	if err != nil {
		return nil, fmt.Errorf("encrypt phone: %w", err)
	}
	row := &store.RespondentRow{
		ID:             uuid.New(),
		TenantID:       tenantID,
		ProjectID:      in.ProjectID,
		PhoneEncrypted: cipher,
		PhoneHash:      hash,
		RegionCode:     region,
		Attributes:     in.Attributes,
		Status:         string(crmapi.RespPending),
		Source:         "imported",
		CreatedAt:      s.deps.Clock().UTC(),
	}
	if err := s.deps.Store.Insert(ctx, row); err != nil {
		return nil, err
	}
	_ = s.deps.Audit.Write(ctx, "respondent.created", auditPayload(actor, map[string]any{
		"respondent_id": row.ID, "project_id": row.ProjectID,
	}))
	return rowToRespDTO(row, MaskPhone(canonical)), nil
```

```go
type ImportTask struct {
	JobID       string
	TenantID    uuid.UUID
	ProjectID   uuid.UUID
	ActorID     uuid.UUID
	Filename    string
	ContentType string
	Body        []byte
	ColumnMap   map[string]string

}

// ImportResult is the outcome of one job run.
type ImportResult struct {
	JobID    string
	Total    int
	Inserted int
	Skipped  int
	Errors   []ImportError
}

const (
	importBodyMax    = 50 * 1024 * 1024 // 50 MB
	importBatchSize  = 1000
)

// Import is the synchronous API: validates input, enqueues an asynq job, returns a ticket.
func (s *RespondentService) Import(ctx context.Context, req crmapi.ImportRequest) (*crmapi.ImportTicket, error) {
	tenantID, actor, err := tenantAndActor(ctx)
	if err != nil {
		return nil, err
	}
	if len(req.Body) == 0 {
		return nil, fmt.Errorf("empty body")
	}
	if len(req.Body) > importBodyMax {
		return nil, crmapi.ErrImportPayloadTooBig
	}
	if req.ColumnMap["phone"] == "" {
		return nil, fmt.Errorf("column_map.phone is required")
	}
	jobID := uuid.NewString()
	payload := struct {
		JobID, Filename, ContentType string
		TenantID, ProjectID, ActorID uuid.UUID
		Body                         []byte
		ColumnMap                    map[string]string
		DefaultAttrs                 map[string]any
	}{
		JobID: jobID, Filename: req.Filename, ContentType: req.ContentType,
		TenantID: tenantID, ProjectID: req.ProjectID, ActorID: actor,
		Body: req.Body, ColumnMap: req.ColumnMap, DefaultAttrs: req.DefaultAttrs,
	}
	enc, err := json.Marshal(&payload)
	if err != nil {
		return nil, err
	}
	taskID, err := s.deps.Async.Enqueue(ctx, crmapi.TaskRespondentImport, enc,
		AsynqOpt{Queue: "crm-imports", MaxRetry: 1, Timeout: 10 * time.Minute})
	if err != nil {
		return nil, err
	}
	_ = s.deps.Audit.Write(ctx, "respondent.import_requested", auditPayload(actor, map[string]any{
		"job_id": jobID, "project_id": req.ProjectID, "filename": req.Filename, "size": len(req.Body),
	}))
	_ = s.deps.Events.Publish(ctx, crmapi.SubjectImportStarted, map[string]any{
		"job_id": jobID, "project_id": req.ProjectID, "tenant_id": tenantID,
	})
	return &crmapi.ImportTicket{
		JobID: taskID, ProjectID: req.ProjectID, StartedAt: s.deps.Clock(),
	}, nil
}

// GetImportStatus reads the progress reporter.
func (s *RespondentService) GetImportStatus(ctx context.Context, jobID string) (*crmapi.ImportStatus, error) {
	if s.deps.Reporter == nil {
		return nil, fmt.Errorf("progress reporter not wired")
	}
	p, err := s.deps.Reporter.Get(ctx, jobID)
	if err != nil {
		return nil, err
	}
	out := &crmapi.ImportStatus{
		JobID: jobID, State: p.State, Total: p.Total, Processed: p.Processed,
		Inserted: p.Inserted, Skipped: p.Skipped, StartedAt: p.StartedAt, FinishedAt: p.FinishedAt,
	}
	for _, e := range p.Errors {
		out.Errors = append(out.Errors, crmapi.ImportError{Row: e.Row, Phone: e.Phone, Message: e.Message})
	}
	return out, nil
}

// RunImport executes one import job — typically called from the asynq handler in cmd/worker.
func (s *RespondentService) RunImport(ctx context.Context, t ImportTask) (*ImportResult, error) {
	rep := t.Reporter
	if rep == nil {
		rep = s.deps.Reporter
	}
	rows, err := s.parseImport(t)
	if err != nil {
		_ = rep.Finish(ctx, t.JobID, ImportProgress{State: "failed", Errors: []ImportError{{Message: err.Error()}}})
		return nil, err
	}
	res := &ImportResult{JobID: t.JobID, Total: len(rows)}
	_ = rep.Update(ctx, t.JobID, ImportProgress{State: "running", Total: len(rows)})

	// Dedup within file by canonical phone.
	seen := make(map[string]struct{}, len(rows))
	batch := make([]*store.RespondentRow, 0, importBatchSize)
	for i, r := range rows {
		canonical, region, err := NormalizePhone(r.phone)
		if err != nil {
			res.Skipped++
			res.Errors = append(res.Errors, ImportError{Row: r.row, Phone: r.phone, Message: "invalid phone"})
			continue
		}
		if _, dup := seen[canonical]; dup {
			res.Skipped++
			res.Errors = append(res.Errors, ImportError{Row: r.row, Phone: canonical, Message: "duplicate in file"})
			continue
		}
		seen[canonical] = struct{}{}
		hash, err := s.deps.Crypto.HashPhone(ctx, canonical)
		if err != nil {
			return res, fmt.Errorf("hash row %d: %w", r.row, err)
		}
		blocked, err := s.deps.DNC.IsBlocked(ctx, t.TenantID, t.ActorID, &t.ProjectID, hash)
		if err != nil {
			return res, fmt.Errorf("dnc lookup row %d: %w", r.row, err)
		}
		if blocked {
			res.Skipped++
			res.Errors = append(res.Errors, ImportError{Row: r.row, Phone: canonical, Message: "in DNC"})
			continue
		}
		cipher, err := s.deps.Crypto.EncryptPhone(ctx, canonical)
		if err != nil {
			return res, fmt.Errorf("encrypt row %d: %w", r.row, err)
		}
		attrs := map[string]any{}
		for k, v := range t.DefaultAttrs {
			attrs[k] = v
		}
		for k, v := range r.attrs {
			attrs[k] = v
		}
		batch = append(batch, &store.RespondentRow{
			ID: uuid.New(), TenantID: t.TenantID, ProjectID: t.ProjectID,
			PhoneEncrypted: cipher, PhoneHash: hash, RegionCode: region,
			Attributes: attrs, Status: string(crmapi.RespPending), Source: "imported",
			CreatedAt: s.deps.Clock().UTC(),
		})
		if len(batch) >= importBatchSize {
			ins, err := s.deps.Store.BulkInsert(ctx, batch)
			if err != nil {
				return res, err
			}
			res.Inserted += ins
			batch = batch[:0]
			_ = rep.Update(ctx, t.JobID, ImportProgress{
				State: "running", Total: len(rows), Processed: i + 1, Inserted: res.Inserted, Skipped: res.Skipped,
			})
		}
	}
	if len(batch) > 0 {
		ins, err := s.deps.Store.BulkInsert(ctx, batch)
		if err != nil {
			return res, err
		}
		res.Inserted += ins
	}
	now := s.deps.Clock()
	final := ImportProgress{
		State: "succeeded", Total: len(rows), Processed: len(rows),
		Inserted: res.Inserted, Skipped: res.Skipped, FinishedAt: &now,
		Errors: res.Errors,
	}
	_ = rep.Finish(ctx, t.JobID, final)
	_ = s.deps.Events.Publish(ctx, crmapi.SubjectImportFinished, map[string]any{
		"job_id": t.JobID, "project_id": t.ProjectID, "inserted": res.Inserted, "skipped": res.Skipped,
	})
	return res, nil
}

// parseImport selects CSV or XLSX based on content-type.
func (s *RespondentService) parseImport(t ImportTask) ([]parsedRow, error) {
	switch {
	case strings.Contains(t.ContentType, "csv") || strings.HasSuffix(strings.ToLower(t.Filename), ".csv"):
		return parseCSV(t.Body, t.ColumnMap)
	case strings.Contains(t.ContentType, "spreadsheetml") || strings.HasSuffix(strings.ToLower(t.Filename), ".xlsx"):
		return parseXLSX(t.Body, t.ColumnMap)
	default:
		return nil, fmt.Errorf("unsupported content-type %q", t.ContentType)
	}
}
```

```go
func parseCSV(body []byte, colmap map[string]string) ([]parsedRow, error) {
	r := csv.NewReader(bytes.NewReader(body))
	r.FieldsPerRecord = -1
	header, err := r.Read()
	if err != nil {
		return nil, fmt.Errorf("read header: %w", err)
	}
	idx := buildHeaderIndex(header, colmap)
	if _, ok := idx["phone"]; !ok {
		return nil, fmt.Errorf("phone column %q not found in header", colmap["phone"])
	}
	var out []parsedRow
	rowNum := 1
	for {
		rec, err := r.Read()
		if err == io.EOF {
			break
		}
		if err != nil {
			return nil, fmt.Errorf("row %d: %w", rowNum+1, err)
		}
		rowNum++
		row := parsedRow{row: rowNum, attrs: map[string]any{}}
		for logical, headerIdx := range idx {
			if headerIdx >= len(rec) {
				continue
			}
			v := strings.TrimSpace(rec[headerIdx])
			if logical == "phone" {
				row.phone = v
			} else if v != "" {
				row.attrs[logical] = v
			}
		}
		out = append(out, row)
	}
	return out, nil
}

// buildHeaderIndex maps logical -> column index in the header.
func buildHeaderIndex(header []string, colmap map[string]string) map[string]int {
	idx := map[string]int{}
	for i, h := range header {
		header[i] = strings.TrimSpace(h)
	}
	for logical, src := range colmap {
		for i, h := range header {
			if strings.EqualFold(h, src) {
				idx[logical] = i
				break
			}
		}
	}
	return idx
}
```

```go
import (
	"bytes"
	"fmt"
	"strings"

	"github.com/xuri/excelize/v2"
)

func parseXLSX(body []byte, colmap map[string]string) ([]parsedRow, error) {
	f, err := excelize.OpenReader(bytes.NewReader(body))
	if err != nil {
		return nil, fmt.Errorf("open xlsx: %w", err)
	}
	defer f.Close()
	sheets := f.GetSheetList()
	if len(sheets) == 0 {
		return nil, fmt.Errorf("no sheets")
	}
	rows, err := f.Rows(sheets[0])
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	var header []string
	var idx map[string]int
	rowNum := 0
	var out []parsedRow
	for rows.Next() {
		rowNum++
		cols, err := rows.Columns()
		if err != nil {
			return nil, fmt.Errorf("row %d: %w", rowNum, err)
		}
		if rowNum == 1 {
			header = make([]string, len(cols))
			for i, c := range cols {
				header[i] = strings.TrimSpace(c)
			}
			idx = buildHeaderIndex(header, colmap)
			if _, ok := idx["phone"]; !ok {
				return nil, fmt.Errorf("phone column %q not found in header", colmap["phone"])
			}
			continue
		}
		row := parsedRow{row: rowNum, attrs: map[string]any{}}
		for logical, hIdx := range idx {
			if hIdx >= len(cols) {
				continue
			}
			v := strings.TrimSpace(cols[hIdx])
			if logical == "phone" {
				row.phone = v
			} else if v != "" {
				row.attrs[logical] = v
			}
		}
		// skip totally-empty rows
		if row.phone == "" && len(row.attrs) == 0 {
			continue
		}
		out = append(out, row)
	}
	return out, rows.Error()
}
```

```go
// RedisProgressReporter persists ImportProgress to a Redis hash and publishes NATS events.
type RedisProgressReporter struct {
	rdb       *redis.Client
	publisher EventPublisher
	ttl       time.Duration
}

// NewRedisProgressReporter wires a reporter.
func NewRedisProgressReporter(rdb *redis.Client, p EventPublisher) *RedisProgressReporter {
	return &RedisProgressReporter{rdb: rdb, publisher: p, ttl: 24 * time.Hour}
}

func (r *RedisProgressReporter) key(jobID string) string { return "crm:import:" + jobID }

// Update writes one snapshot.
func (r *RedisProgressReporter) Update(ctx context.Context, jobID string, p ImportProgress) error {
	enc, err := json.Marshal(p)
	if err != nil {
		return err
	}
	if err := r.rdb.Set(ctx, r.key(jobID), enc, r.ttl).Err(); err != nil {
		return err
	}
	if r.publisher != nil {
		_ = r.publisher.Publish(ctx, "crm.respondents.import.progress", p)
	}
	return nil
}

// Finish writes a terminal snapshot.
func (r *RedisProgressReporter) Finish(ctx context.Context, jobID string, p ImportProgress) error {
	if p.FinishedAt == nil {
		now := time.Now()
		p.FinishedAt = &now
	}
	return r.Update(ctx, jobID, p)
}

// Get reads the snapshot.
func (r *RedisProgressReporter) Get(ctx context.Context, jobID string) (*ImportProgress, error) {
	raw, err := r.rdb.Get(ctx, r.key(jobID)).Bytes()
	if err == redis.Nil {
		return nil, fmt.Errorf("import job %q not found", jobID)
	}
	if err != nil {
		return nil, err
	}
	var p ImportProgress
	if err := json.Unmarshal(raw, &p); err != nil {
		return nil, err
	}
	return &p, nil
}
```

```go
		hash, err = s.deps.Crypto.HashPhone(ctx, canonical)
		if err != nil {
			return nil, err
		}
	}
	rows, total, err := s.deps.Store.Search(ctx, tenantID, f, hash)
	if err != nil {
		return nil, err
	}
	out := &crmapi.SearchRespondentsResult{Items: make([]crmapi.Respondent, len(rows)), TotalCount: total}
	for i := range rows {
		// In list responses the phone is not decrypted (cost). UI displays a placeholder mask.
		out.Items[i] = *rowToRespDTO(&rows[i], "+7-***-***-**-**")
	}
	return out, nil
}

// Delete starts the 30-day deletion grace period (FR-§13.3).
func (s *RespondentService) Delete(ctx context.Context, id uuid.UUID) (*crmapi.DeletionRequest, error) {
	tenantID, actor, err := tenantAndActor(ctx)
	if err != nil {
		return nil, err
	}
	deleteAt := s.deps.Clock().UTC().Add(30 * 24 * time.Hour)
	if err := s.deps.Store.MarkDeletionRequested(ctx, tenantID, id, deleteAt); err != nil {
		if errors.Is(err, store.ErrNoRows) {
			return nil, crmapi.ErrRespondentNotFound
		}
		return nil, err
	}
	_ = s.deps.Audit.Write(ctx, "respondent.deletion_requested", auditPayload(actor, map[string]any{
		"respondent_id": id, "delete_at": deleteAt,
	}))
	_ = s.deps.Events.Publish(ctx, crmapi.SubjectRespondentDelete, map[string]any{
		"respondent_id": id, "tenant_id": tenantID, "delete_at": deleteAt,
	})
	return &crmapi.DeletionRequest{RespondentID: id, DeleteAt: deleteAt}, nil
}
```

```go
package service

import (
	"context"

	"github.com/hibiken/asynq"

	crmapi "github.com/sociopulse/platform/internal/crm/api"
)

const purgeBatch = 1000

// RunPurge sweeps deletion-requested respondents whose grace period has elapsed.
func (s *RespondentService) RunPurge(ctx context.Context) (int, error) {
	total := 0
	for {
		n, err := s.deps.Store.PurgeDue(ctx, s.deps.Clock().UTC(), purgeBatch)
		if err != nil {
			return total, err
		}
		if n == 0 {
			return total, nil
		}
		total += n
		_ = s.deps.Audit.Write(ctx, "respondent.purged", map[string]any{"count": n})
		if n < purgeBatch {
			return total, nil
		}
	}
}

// PurgeHandler returns an asynq handler that drives RunPurge.
func PurgeHandler(s *RespondentService) func(ctx context.Context, t *asynq.Task) error {
	return func(ctx context.Context, _ *asynq.Task) error {
		_, err := s.RunPurge(ctx)
		return err
	}
}
```

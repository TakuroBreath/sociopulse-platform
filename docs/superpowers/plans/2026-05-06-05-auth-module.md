# Auth Module Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. REQUIRED SUB-SKILL: superpowers:test-driven-development for every task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the production-ready `internal/auth` module providing password authentication, JWT issuance/validation, refresh-token rotation with revocation, TOTP 2FA, brute-force protection, RBAC enforcement, user CRUD, and a thin HTTP layer mounted on the gateway. Replace the stub `Auth` middleware shipped in Plan 02 with the real one. Coverage of `internal/auth/service/` ≥ 90%; integration tests run against PostgreSQL + Redis containers brought up in Plan 03.

**Architecture:** Hexagonal layout per repo convention.

- `internal/auth/api/` — public Go interfaces (`Authenticator`, `UserService`, `RBACChecker`, `SessionRevoker`) and DTOs (`Claims`, `AuthResult`, `User`, `LoginRequest`, …). Imported by other modules and by `cmd/api`.
- `internal/auth/service/` — business logic implementations: password hashing, JWT issuer/validator, login/refresh/logout flows, rate limiting, lockout, TOTP enrollment, RBAC matrix.
- `internal/auth/store/` — persistence: `UserStore` (Postgres via pgx), `SessionStore` / `RefreshStore` (Redis), `LoginAttemptStore` (Redis sliding window), `TOTPStore` (Postgres, secret encrypted via tenancy KMS).
- `internal/auth/transport/http/` — chi router mounted under `/api/`; handlers are thin adapters that translate HTTP into service calls.
- `internal/auth/events/` — emits `audit.event` to NATS via the audit module's Publisher (provided through DI in Plan 04).
- `pkg/passwords/` — reusable Argon2id hash/verify helper, no module-specific dependencies.
- `pkg/middleware/auth/` — chi middleware that consumes `api.Authenticator` and enriches `RequestContext` with `Claims`.

```go
// Package passwords provides Argon2id-based password hashing.
//
// The encoded form follows the PHC string format:
//   $argon2id$v=19$m=65536,t=3,p=4$<salt-b64>$<hash-b64>
//
// Defaults match §14.2 of the СоциоПульс system-design spec:
//   memory      = 64 * 1024 KiB
//   iterations  = 3
//   parallelism = 4
//   salt length = 16 bytes
//   key length  = 32 bytes
package passwords

// Params controls Argon2id cost and output size. Zero values are invalid;
// always start from DefaultParams() and tune from there.
type Params struct {
    Memory      uint32 // KiB
    Iterations  uint32
    Parallelism uint8
    SaltLength  uint32 // bytes
    KeyLength   uint32 // bytes
}

// DefaultParams returns the production defaults.
func DefaultParams() Params {
    return Params{
        Memory:      64 * 1024,
        Iterations:  3,
        Parallelism: 4,
        SaltLength:  16,
        KeyLength:   32,
    }
}

// Validate returns nil if all fields are within sane bounds.
func (p Params) Validate() error { /* see Step 3 */ }
```

```go
package passwords

import (
    "crypto/rand"
    "crypto/subtle"
    "errors"
    "fmt"

    "golang.org/x/crypto/argon2"
)

var (
    ErrInvalidHash         = errors.New("passwords: invalid hash format")
    ErrIncompatibleVersion = errors.New("passwords: incompatible argon2 version")
)

// Hash returns a PHC-encoded Argon2id hash of password using p.
func Hash(password string, p Params) (string, error) {
    if err := p.Validate(); err != nil {
        return "", err
    }
    salt := make([]byte, p.SaltLength)
    if _, err := rand.Read(salt); err != nil {
        return "", fmt.Errorf("passwords: read salt: %w", err)
    }
    key := argon2.IDKey([]byte(password), salt, p.Iterations, p.Memory, p.Parallelism, p.KeyLength)
    return encode(p, salt, key), nil
}

// Verify reports whether password matches the encoded PHC hash.
// It is constant-time with respect to a hash of the right shape.
func Verify(encoded, password string) (bool, error) {
    p, salt, key, err := decode(encoded)
    if err != nil {
        return false, err
    }
    candidate := argon2.IDKey([]byte(password), salt, p.Iterations, p.Memory, p.Parallelism, uint32(len(key)))
    return subtle.ConstantTimeCompare(key, candidate) == 1, nil
}
```

- [ ] **Step 5: Implement PHC encode/decode in `encoding.go`**

```go
package passwords

import (
    "encoding/base64"
    "fmt"
    "strings"
)

const phcPrefix = "$argon2id$v=19$"

func encode(p Params, salt, key []byte) string {
    return fmt.Sprintf(
        "%sm=%d,t=%d,p=%d$%s$%s",
        phcPrefix,
        p.Memory, p.Iterations, p.Parallelism,
        base64.RawStdEncoding.EncodeToString(salt),
        base64.RawStdEncoding.EncodeToString(key),
    )
}

func decode(s string) (Params, []byte, []byte, error) {
    if !strings.HasPrefix(s, phcPrefix) {
        return Params{}, nil, nil, ErrInvalidHash
    }
    rest := strings.TrimPrefix(s, phcPrefix)
    parts := strings.Split(rest, "$")
    if len(parts) != 3 {
        return Params{}, nil, nil, ErrInvalidHash
    }
    var p Params
    if _, err := fmt.Sscanf(parts[0], "m=%d,t=%d,p=%d", &p.Memory, &p.Iterations, &p.Parallelism); err != nil {
        return Params{}, nil, nil, fmt.Errorf("%w: params: %v", ErrInvalidHash, err)
    }
    salt, err := base64.RawStdEncoding.DecodeString(parts[1])
    if err != nil {
        return Params{}, nil, nil, fmt.Errorf("%w: salt: %v", ErrInvalidHash, err)
    }
    key, err := base64.RawStdEncoding.DecodeString(parts[2])
    if err != nil {
        return Params{}, nil, nil, fmt.Errorf("%w: key: %v", ErrInvalidHash, err)
    }
    p.SaltLength = uint32(len(salt))
    p.KeyLength = uint32(len(key))
    return p, salt, key, nil
}
```

```go
// Hasher is the interface implemented by Default. Tests in dependent packages
// can substitute a fake (e.g. SHA-1 of password) so unit tests don't pay the

type defaultHasher struct{ p Params }
```

```go
package service

import (
    "crypto/rand"
    "encoding/hex"
    "errors"
    "fmt"
    "time"

    "github.com/golang-jwt/jwt/v5"
    "github.com/google/uuid"

    authapi "sociopulse/internal/auth/api"
)

const (
    accessTokenType  = "access"
    refreshTokenType = "refresh"
)

// JWTConfig comes from §14.2 auth.jwt block.
type JWTConfig struct {
    Issuer       string
    Secret       []byte // sourced from Lockbox via env JWT_SIGNING_SECRET
    AccessTTL    time.Duration
    RefreshTTL   time.Duration
    LeewaySecond int
}

type JWTIssuer struct {
    cfg JWTConfig
    now func() time.Time
}

func NewJWTIssuer(cfg JWTConfig, clock func() time.Time) *JWTIssuer {
    if clock == nil {
        clock = time.Now
    }
    return &JWTIssuer{cfg: cfg, now: clock}
}

func (j *JWTIssuer) IssueAccess(c authapi.Claims) (string, time.Time, error) {
    return j.sign(c, accessTokenType, j.cfg.AccessTTL)
}

func (j *JWTIssuer) IssueRefresh(c authapi.Claims) (string, time.Time, error) {
    return j.sign(c, refreshTokenType, j.cfg.RefreshTTL)
}

func (j *JWTIssuer) sign(c authapi.Claims, typ string, ttl time.Duration) (string, time.Time, error) {
    if c.UserID == uuid.Nil || c.TenantID == uuid.Nil {
        return "", time.Time{}, errors.New("jwt: user_id and tenant_id required")
    }
    now := j.now()
    exp := now.Add(ttl)
    if c.JTI == "" {
        c.JTI = randomJTI()
    }
    if c.SessionID == "" {
        c.SessionID = randomJTI()
    }
    rolesStr := make([]string, len(c.Roles))
    for i, r := range c.Roles {
        rolesStr[i] = string(r)
    }
    claims := jwt.MapClaims{
        "iss":       j.cfg.Issuer,
        "sub":       c.UserID.String(),
        "tid":       c.TenantID.String(),
        "login":     c.Login,
        "roles":     rolesStr,
        "sid":       c.SessionID,
        "jti":       c.JTI,
        "iat":       now.Unix(),
        "exp":       exp.Unix(),
        "typ":       typ,
        "totp_done": c.TOTPDone,
    }
    tok := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    s, err := tok.SignedString(j.cfg.Secret)
    if err != nil {
        return "", time.Time{}, fmt.Errorf("jwt: sign: %w", err)
    }
    return s, exp, nil
}

func (j *JWTIssuer) Validate(token string, expectedType string) (authapi.Claims, error) {
    parser := jwt.NewParser(
        jwt.WithValidMethods([]string{jwt.SigningMethodHS256.Alg()}),
        jwt.WithLeeway(time.Duration(j.cfg.LeewaySecond)*time.Second),
        jwt.WithIssuer(j.cfg.Issuer),
    )
    parsed, err := parser.Parse(token, func(t *jwt.Token) (any, error) { return j.cfg.Secret, nil })
    if err != nil {
        return authapi.Claims{}, fmt.Errorf("%w: %v", authapi.ErrTokenInvalid, err)
    }
    mc, ok := parsed.Claims.(jwt.MapClaims)
    if !ok {
        return authapi.Claims{}, authapi.ErrTokenInvalid
    }
    if typ, _ := mc["typ"].(string); typ != expectedType {
        return authapi.Claims{}, authapi.ErrTokenInvalid
    }
    return mapClaimsTo(mc)
}

func randomJTI() string {
    var b [16]byte
    _, _ = rand.Read(b[:])
    return hex.EncodeToString(b[:])
}

// mapClaimsTo decodes jwt.MapClaims into the typed authapi.Claims (omitted here for
// brevity — see Step 6 for the full implementation including roles/uuid parsing).
```

```go
import (
    "crypto/rand"
    "encoding/binary"
    "errors"
)

const tempPasswordAlphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-_"

func GenerateTempPassword() (string, error) {
    if len(tempPasswordAlphabet) != 64 {
        return "", errors.New("alphabet must be 64 chars for unbiased mod")
    }
    var raw [16]byte
    if _, err := rand.Read(raw[:]); err != nil {
        return "", err
    }
    out := make([]byte, 16)
    for i, b := range raw {
        out[i] = tempPasswordAlphabet[b%64]
    }
    _ = binary.BigEndian // silence import for variants
    return string(out), nil
}
```

- [ ] **Step 3: Refresh-token store contract (`internal/auth/store/refresh_store.go`)**

We store, in Redis:
- `refresh:<jti>` → JSON `{user_id, tenant_id, sid, exp}`. TTL = `refresh_ttl`. Acts as a whitelist of currently-valid refresh tokens.
- `refresh:rotated:<jti>` → `<new_jti>`. TTL = 30 d. Used for replay detection: if an already-rotated `jti` is presented, every descendant in the chain is revoked (refresh-rotation reuse detection).
- `revoke:session:<sid>` → `1`. TTL = `refresh_ttl`. Used by `IsRevoked`.
- `revoke:user:<user_id>:cutoff` → unix-timestamp. Anything issued before that time is revoked. Used for `RevokeAllForUser`.

```go
package store

import (
    "context"
    "encoding/json"
    "errors"
    "fmt"
    "time"

    "github.com/google/uuid"
    "github.com/redis/go-redis/v9"
)

type RefreshRecord struct {
    UserID    uuid.UUID `json:"user_id"`
    TenantID  uuid.UUID `json:"tenant_id"`
    SessionID string    `json:"sid"`
    ExpiresAt time.Time `json:"exp"`
}

type RefreshStore struct {
    rdb *redis.Client
    ttl time.Duration
}

func NewRefreshStore(rdb *redis.Client, ttl time.Duration) *RefreshStore {
    return &RefreshStore{rdb: rdb, ttl: ttl}
}

func (s *RefreshStore) Save(ctx context.Context, jti string, rec RefreshRecord) error {
    b, _ := json.Marshal(rec)
    return s.rdb.Set(ctx, refreshKey(jti), b, s.ttl).Err()
}

func (s *RefreshStore) Lookup(ctx context.Context, jti string) (RefreshRecord, error) {
    v, err := s.rdb.Get(ctx, refreshKey(jti)).Bytes()
    if errors.Is(err, redis.Nil) {
        return RefreshRecord{}, ErrRefreshNotFound
    }
    if err != nil {
        return RefreshRecord{}, fmt.Errorf("refresh lookup: %w", err)
    }
    var rec RefreshRecord
    if err := json.Unmarshal(v, &rec); err != nil {
        return RefreshRecord{}, fmt.Errorf("refresh decode: %w", err)
    }
    return rec, nil
}

func (s *RefreshStore) Rotate(ctx context.Context, oldJTI, newJTI string, rec RefreshRecord) (rotated bool, err error) {
    // Atomic swap via Lua: if rotated:<old> already exists -> reuse; mark whole chain.
    script := redis.NewScript(`
        local prev = redis.call("GET", KEYS[1])
        if prev then
            return {0, prev}
        end
        redis.call("SET", KEYS[1], ARGV[1], "EX", ARGV[2])
        redis.call("DEL", KEYS[2])
        redis.call("SET", KEYS[3], ARGV[3], "EX", ARGV[2])
        return {1, ""}
    `)
    keys := []string{
        rotatedKey(oldJTI),
        refreshKey(oldJTI),
        refreshKey(newJTI),
    }
    payload, _ := json.Marshal(rec)
    res, err := script.Run(ctx, s.rdb, keys, newJTI, int(s.ttl.Seconds()), payload).Result()
    if err != nil {
        return false, err
    }
    arr, _ := res.([]any)
    ok, _ := arr[0].(int64)
    if ok == 0 {
        // Reuse detected. Caller should revoke session.
        return false, ErrRefreshAlreadyRotated
    }
    return true, nil
}

func (s *RefreshStore) Delete(ctx context.Context, jti string) error {
    return s.rdb.Del(ctx, refreshKey(jti)).Err()
}

var (
    ErrRefreshNotFound       = errors.New("refresh: not found")
    ErrRefreshAlreadyRotated = errors.New("refresh: already rotated (reuse)")
)

func refreshKey(jti string) string { return "auth:refresh:" + jti }
func rotatedKey(jti string) string { return "auth:refresh:rotated:" + jti }
```

```go
package service

import (
    "context"
    "fmt"
    "strconv"
    "time"

    "github.com/google/uuid"
    "github.com/redis/go-redis/v9"

    authapi "sociopulse/internal/auth/api"
)

type SessionRevoker struct {
    rdb *redis.Client
    ttl time.Duration
}

func NewSessionRevoker(rdb *redis.Client, ttl time.Duration) *SessionRevoker {
    return &SessionRevoker{rdb: rdb, ttl: ttl}
}

func (r *SessionRevoker) RevokeSession(ctx context.Context, sid string) error {
    return r.rdb.Set(ctx, sidKey(sid), "1", r.ttl).Err()
}

func (r *SessionRevoker) RevokeAllForUser(ctx context.Context, userID uuid.UUID) error {
    return r.rdb.Set(ctx, userCutoffKey(userID), strconv.FormatInt(time.Now().Unix(), 10), r.ttl).Err()
}

func (r *SessionRevoker) IsRevoked(ctx context.Context, c authapi.Claims) (bool, error) {
    if c.SessionID != "" {
        if exists, err := r.rdb.Exists(ctx, sidKey(c.SessionID)).Result(); err == nil && exists > 0 {
            return true, nil
        }
    }
    cutoff, err := r.rdb.Get(ctx, userCutoffKey(c.UserID)).Int64()
    if err == nil && c.IssuedAt.Unix() < cutoff {
        return true, nil
    }
    if err != nil && err != redis.Nil {
        return false, fmt.Errorf("revoker: %w", err)
    }
    return false, nil
}

func sidKey(sid string) string             { return "auth:revoke:sid:" + sid }
func userCutoffKey(id uuid.UUID) string    { return "auth:revoke:user:" + id.String() + ":cutoff" }
```

```go
type Authenticator struct {

    tenants    TenantResolver       // org_id -> tenant_id
    hasher     passwords.Hasher
    issuer     authapi.JWTIssuer
    revoker    authapi.SessionRevoker
    refreshes  RefreshTokenStore
    rate       RateLimiter
    lockout    Lockout
    totp       TOTPVerifier
    auditor    auditapi.Publisher
    metrics    *Metrics
    clock      Clock
}

type TenantResolver interface {
    ResolveByOrgID(ctx context.Context, orgID string) (uuid.UUID, error)
}

type RefreshTokenStore interface {
    Save(ctx context.Context, jti string, rec store.RefreshRecord) error
    Lookup(ctx context.Context, jti string) (store.RefreshRecord, error)
    Rotate(ctx context.Context, oldJTI, newJTI string, rec store.RefreshRecord) (bool, error)
    Delete(ctx context.Context, jti string) error
}

type RateLimiter interface {
    AllowIP(ctx context.Context, ip netip.Addr) (bool, error)
    AllowAccount(ctx context.Context, userID uuid.UUID) (bool, error)
}

type Lockout interface {
    IsLocked(ctx context.Context, userID uuid.UUID) (bool, error)
    RegisterFailure(ctx context.Context, userID uuid.UUID) (locked bool, err error)
    Reset(ctx context.Context, userID uuid.UUID) error
}

type TOTPVerifier interface {
    Verify(ctx context.Context, userID uuid.UUID, code string) (bool, error)
}
```

Algorithm for `Login`:

```
1.  IP-rate-limit:    rate.AllowIP() — if false, audit + return ErrRateLimitExceeded
2.  tenant := tenants.ResolveByOrgID(orgID)
       on error -> audit unknown_org, return ErrInvalidCredentials (timing-safe dummy verify)
3.  user, err := users.GetByLogin(ctx, tenant, login)
       on error or err.Is(NotFound) -> dummy verify, audit, ErrInvalidCredentials
4.  if user.archived_at != nil -> ErrAccountArchived
5.  if lockout.IsLocked(user.id) -> ErrAccountLocked (no verify, but still spend time on dummy)
6.  account-rate-limit: rate.AllowAccount(user.id) -> if false, ErrRateLimitExceeded
7.  ok, _ := hasher.Verify(user.password_hash, password)
8.  if !ok:
        lockout.RegisterFailure(user.id)
        audit auth.login.failure
        metrics.LoginFailures.Inc()
        return ErrInvalidCredentials
9.  lockout.Reset(user.id)
10. if user.must_change_pwd -> partial-token typ=password-reset, return ErrPasswordExpired wrapped
11. if user.totp_enabled:
        partial = issuer.IssueAccess(claims with typ=partial, ttl=5min, totp_done=false)
        return AuthResult{TOTPRequired: true, AccessToken: partial.Token, ...}
12. issue access+refresh, save refresh, audit success, metrics.LoginSuccess.Inc()
```

Algorithm for `Refresh`:

```
1.  claims := issuer.Validate(token, "refresh")  -> ErrTokenInvalid maps to caller
2.  rec, err := refreshes.Lookup(claims.JTI)     -> if NotFound -> ErrTokenInvalid
3.  newJTI := randomJTI()
4.  rotated, err := refreshes.Rotate(claims.JTI, newJTI, rec)
        if errors.Is(err, ErrRefreshAlreadyRotated):
            revoker.RevokeSession(claims.SID)
            audit auth.refresh.replay
            return ErrTokenInvalid
5.  newClaims := claims; newClaims.JTI = newJTI
6.  issuer.IssueAccess + issuer.IssueRefresh(newClaims) — refresh keeps SID, new JTI
7.  audit auth.refresh.success
```

Algorithm for `Logout`:

```
1.  claims := issuer.Validate(refreshToken, "refresh") — ignore "expired" error
2.  refreshes.Delete(claims.JTI)
3.  revoker.RevokeSession(claims.SID)
4.  audit auth.logout
```

```go
package service

import (
    "context"
    "fmt"
    "net/netip"
    "time"

    "github.com/google/uuid"
    "github.com/redis/go-redis/v9"
)

type RateLimiterRedis struct {
    rdb            *redis.Client
    perIPPerHour   int
    perUserPerHour int
    window         time.Duration // = 1 hour
    clock          Clock
}

func (r *RateLimiterRedis) AllowIP(ctx context.Context, ip netip.Addr) (bool, error) {
    return r.allow(ctx, "auth:rl:ip:"+ip.String(), r.perIPPerHour)
}

func (r *RateLimiterRedis) AllowAccount(ctx context.Context, userID uuid.UUID) (bool, error) {
    return r.allow(ctx, "auth:rl:user:"+userID.String(), r.perUserPerHour)
}

func (r *RateLimiterRedis) allow(ctx context.Context, key string, max int) (bool, error) {
    now := r.clock.Now()
    minScore := float64(now.Add(-r.window).UnixMilli())
    nowScore := float64(now.UnixMilli())
    member := fmt.Sprintf("%d-%s", now.UnixNano(), randomJTI()[:8])
    pipe := r.rdb.Pipeline()
    pipe.ZRemRangeByScore(ctx, key, "0", fmt.Sprintf("(%f", minScore))
    countCmd := pipe.ZCard(ctx, key)
    pipe.ZAdd(ctx, key, redis.Z{Score: nowScore, Member: member})
    pipe.Expire(ctx, key, r.window+time.Minute)
    if _, err := pipe.Exec(ctx); err != nil {
        return false, err
    }
    return countCmd.Val() < int64(max), nil
}
```

The trick: we run `ZRemRangeByScore` and `ZCard` and `ZAdd` in one pipeline; the value of `ZCard` is read *before* the new element is added, so the comparison is correct. (The new attempt is always recorded so that a flood is correctly rate-limited even if the request is later rejected — every attempt counts.)

```go
package service

import (
    "context"
    "errors"
    "strconv"
    "time"

    "github.com/google/uuid"
    "github.com/redis/go-redis/v9"
)

type LockoutRedis struct {
    rdb       *redis.Client
    threshold int
    duration  time.Duration
    clock     Clock
}

func (l *LockoutRedis) IsLocked(ctx context.Context, id uuid.UUID) (bool, error) {
    v, err := l.rdb.Get(ctx, untilKey(id)).Int64()
    if errors.Is(err, redis.Nil) {
        return false, nil
    }
    if err != nil {
        return false, err
    }
    return l.clock.Now().Unix() < v, nil
}

func (l *LockoutRedis) RegisterFailure(ctx context.Context, id uuid.UUID) (bool, error) {
    n, err := l.rdb.Incr(ctx, failsKey(id)).Result()
    if err != nil {
        return false, err
    }
    if n == 1 {
        l.rdb.Expire(ctx, failsKey(id), l.duration*2)
    }
    if n >= int64(l.threshold) {
        until := l.clock.Now().Add(l.duration).Unix()
        if err := l.rdb.Set(ctx, untilKey(id), strconv.FormatInt(until, 10), l.duration).Err(); err != nil {
            return false, err
        }
        return true, nil
    }
    return false, nil
}

func (l *LockoutRedis) Reset(ctx context.Context, id uuid.UUID) error {
    return l.rdb.Del(ctx, failsKey(id), untilKey(id)).Err()
}

func failsKey(id uuid.UUID) string { return "auth:lk:fails:" + id.String() }
func untilKey(id uuid.UUID) string { return "auth:lk:until:" + id.String() }
```

```go
package service

import (
    "bytes"
    "context"
    "encoding/base32"
    "errors"
    "fmt"
    "image/png"
    "time"

    "github.com/google/uuid"
    "github.com/pquerna/otp"
    "github.com/pquerna/otp/totp"

    auditapi "sociopulse/internal/audit/api"
    authapi "sociopulse/internal/auth/api"
    "sociopulse/internal/tenancy/api"
)

type TOTPService struct {
    issuer     string // "SocioPulse"
    period     uint
    digits     otp.Digits
    store      TOTPStore
    users      UserTOTPUpdater
    kms        api.KMSResolver
    auditor    auditapi.Publisher
    clock      Clock
}

type TOTPStore interface {
    Upsert(ctx context.Context, userID, tenantID uuid.UUID, encSecret []byte, dekID string) error
    Get(ctx context.Context, userID uuid.UUID) (encSecret []byte, dekID string, enrolled bool, err error)
    Confirm(ctx context.Context, userID uuid.UUID, at time.Time) error
    Delete(ctx context.Context, userID uuid.UUID) error
    UpdateLastUsed(ctx context.Context, userID uuid.UUID, at time.Time) error
}

type UserTOTPUpdater interface {
    UserLogin(ctx context.Context, id uuid.UUID) (login, tenantID string, err error)
    SetTOTPEnabled(ctx context.Context, id uuid.UUID, enabled bool) error
}

func (t *TOTPService) StartEnrollment(ctx context.Context, userID uuid.UUID) (authapi.TOTPEnrollment, error) {
    login, tenantStr, err := t.users.UserLogin(ctx, userID)
    if err != nil {
        return authapi.TOTPEnrollment{}, err
    }
    tenantID, _ := uuid.Parse(tenantStr)
    key, err := totp.Generate(totp.GenerateOpts{
        Issuer:      t.issuer,
        AccountName: login,
        Period:      t.period,
        Digits:      t.digits,
    })
    if err != nil {
        return authapi.TOTPEnrollment{}, fmt.Errorf("totp gen: %w", err)
    }
    secret := key.Secret() // base32
    enc, dekID, err := t.kms.Encrypt(ctx, tenantID, []byte(secret))
    if err != nil {
        return authapi.TOTPEnrollment{}, fmt.Errorf("totp encrypt: %w", err)
    }
    if err := t.store.Upsert(ctx, userID, tenantID, enc, dekID); err != nil {
        return authapi.TOTPEnrollment{}, err
    }
    img, err := key.Image(200, 200)
    if err != nil {
        return authapi.TOTPEnrollment{}, fmt.Errorf("totp qr: %w", err)
    }
    var pngBuf bytes.Buffer
    if err := png.Encode(&pngBuf, img); err != nil {
        return authapi.TOTPEnrollment{}, fmt.Errorf("totp png: %w", err)
    }
    return authapi.TOTPEnrollment{
        Secret:    secret,
        URI:       key.URL(),
        QRCodePNG: pngBuf.Bytes(),
    }, nil
}

func (t *TOTPService) ConfirmEnrollment(ctx context.Context, userID uuid.UUID, code string) error {
    secret, _, err := t.fetchSecret(ctx, userID)
    if err != nil {
        return err
    }
    valid, err := totp.ValidateCustom(code, secret, t.clock.Now(), totp.ValidateOpts{
        Period: t.period, Skew: 1, Digits: t.digits,
    })
    if err != nil || !valid {
        return authapi.ErrTOTPInvalid
    }
    if err := t.store.Confirm(ctx, userID, t.clock.Now()); err != nil {
        return err
    }
    if err := t.users.SetTOTPEnabled(ctx, userID, true); err != nil {
        return err
    }
    t.audit(ctx, "auth.totp.enrolled", userID)
    return nil
}

func (t *TOTPService) Verify(ctx context.Context, userID uuid.UUID, code string) (bool, error) {
    secret, enrolled, err := t.fetchSecret(ctx, userID)
    if err != nil {
        return false, err
    }
    if !enrolled {
        return false, authapi.ErrTOTPInvalid
    }
    ok, _ := totp.ValidateCustom(code, secret, t.clock.Now(), totp.ValidateOpts{
        Period: t.period, Skew: 1, Digits: t.digits,
    })
    if ok {
        _ = t.store.UpdateLastUsed(ctx, userID, t.clock.Now())
    }
    return ok, nil
}

func (t *TOTPService) Disable(ctx context.Context, userID uuid.UUID) error {
    if err := t.store.Delete(ctx, userID); err != nil {
        return err
    }
    if err := t.users.SetTOTPEnabled(ctx, userID, false); err != nil {
        return err
    }
    t.audit(ctx, "auth.totp.disabled", userID)
    return nil
}

func (t *TOTPService) fetchSecret(ctx context.Context, userID uuid.UUID) (string, bool, error) {
    enc, dekID, enrolled, err := t.store.Get(ctx, userID)
    if err != nil {
        return "", false, err
    }
    plaintext, err := t.kms.Decrypt(ctx, dekID, enc)
    if err != nil {
        return "", enrolled, fmt.Errorf("totp decrypt: %w", err)
    }
    return string(plaintext), enrolled, nil
}
```
